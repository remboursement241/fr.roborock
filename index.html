const slides = document.querySelector('.slides'); const indicators = document.querySelectorAll('.indicators span'); const prev = document.getElementById('prev'); const next = document.getElementById('next'); let currentIndex = 0; function updateSlider(index) { slides.style.transform = translateX(-${index * 100}%); indicators.forEach((indicator, i) => { indicator.classList.toggle('active', i === index); }); } prev.addEventListener('click', () => { currentIndex = (currentIndex === 0) ? indicators.length - 1 : currentIndex - 1; updateSlider(currentIndex); }); next.addEventListener('click', () => { currentIndex = (currentIndex === indicators.length - 1) ? 0 : currentIndex + 1; updateSlider(currentIndex); }); indicators.forEach((indicator, i) => { indicator.addEventListener('click', () => { currentIndex = i; updateSlider(currentIndex); }); });
